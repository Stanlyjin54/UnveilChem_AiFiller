# UnveilChem AI文档智能解析与参数识别系统
# 源代码文档

## 1. 项目概述

UnveilChem是一款基于人工智能的化工文档智能解析与参数识别系统。该系统采用深度学习、自然语言处理和计算机视觉技术，能够自动识别和提取化工文档中的工艺参数，并实现与Aspen Plus等化工软件的无缝对接。

### 1.1 技术架构
- **前端**：React + TypeScript + Ant Design
- **后端**：FastAPI + SQLAlchemy + Python 3.8+
- **AI引擎**：PyTorch + Transformers + OpenCV
- **文档解析**：Apache Tika + Tesseract OCR
- **数据库**：SQLite（开发）/ PostgreSQL（生产）

### 1.2 核心功能模块
1. 文档上传与管理模块
2. AI文档解析引擎
3. 参数提取与验证模块
4. 软件对接适配器
5. 用户管理与权限控制
6. 数据分析与可视化模块

### 2.2 文档上传组件 (DocumentUpload.tsx)
```typescript
import React, { useState } from 'react';
import { Upload, Button, message, Progress, Card } from 'antd';
import { InboxOutlined, UploadOutlined } from '@ant-design/icons';
import type { UploadFile } from 'antd/es/upload';
import { uploadDocument } from '../services/documentService';

const { Dragger } = Upload;

const DocumentUpload: React.FC = () => {
  const [fileList, setFileList] = useState<UploadFile[]>([]);
  const [uploading, setUploading] = useState(false);
  const [progress, setProgress] = useState(0);

  const props = {
    name: 'file',
    multiple: true,
    fileList,
    beforeUpload: (file: File) => {
      const isValidFormat = [
        'application/pdf',
        'application/vnd.openxmlformats-officedocument.wordprocessingml.document',
        'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet',
        'image/jpeg',
        'image/png'
      ].includes(file.type);
      
      if (!isValidFormat) {
        message.error(`${file.name} 格式不支持`);
        return false;
      }
      
      const isLt100M = file.size / 1024 / 1024 < 100;
      if (!isLt100M) {
        message.error(`${file.name} 文件大小超过100MB限制`);
        return false;
      }
      
      return true;
    },
    onChange: (info: any) => {
      setFileList(info.fileList);
    },
    onDrop: (e: React.DragEvent) => {
      console.log('Dropped files', e.dataTransfer.files);
    },
  };

  const handleUpload = async () => {
    if (fileList.length === 0) {
      message.warning('请选择要上传的文档');
      return;
    }

    setUploading(true);
    setProgress(0);

    try {
      const formData = new FormData();
      fileList.forEach(file => {
        formData.append('files', file.originFileObj);
      });

      const result = await uploadDocument(formData, (progress) => {
        setProgress(progress);
      });

      message.success('文档上传成功');
      setFileList([]);
    } catch (error) {
      message.error('文档上传失败');
      console.error('Upload error:', error);
    } finally {
      setUploading(false);
      setProgress(0);
    }
  };

  return (
    <Card title="文档上传" style={{ marginBottom: 16 }}>
      <Dragger {...props} disabled={uploading}>
        <p className="ant-upload-drag-icon">
          <InboxOutlined />
        </p>
        <p className="ant-upload-text">点击或拖拽文件到此区域上传</p>
        <p className="ant-upload-hint">
          支持 PDF、Word、Excel、图片等格式，单个文件不超过100MB
        </p>
      </Dragger>
      
      {uploading && (
        <div style={{ marginTop: 16 }}>
          <Progress percent={progress} status="active" />
        </div>
      )}
      
      <div style={{ marginTop: 16, textAlign: 'right' }}>
        <Button
          type="primary"
          icon={<UploadOutlined />}
          onClick={handleUpload}
          loading={uploading}
          disabled={fileList.length === 0}
        >
          开始上传
        </Button>
      </div>
    </Card>
  );
};

export default DocumentUpload;
```

### 2.3 参数提取组件 (ParameterExtraction.tsx)
```typescript
import React, { useState, useEffect } from 'react';
import { Table, Card, Tag, Button, Space, Input, Select } from 'antd';
import { EditOutlined, CheckOutlined, CloseOutlined } from '@ant-design/icons';
import { getExtractedParameters, updateParameter } from '../services/parameterService';
import type { Parameter } from '../types/parameter';

const { Search } = Input;
const { Option } = Select;

const ParameterExtraction: React.FC = () => {
  const [parameters, setParameters] = useState<Parameter[]>([]);
  const [loading, setLoading] = useState(false);
  const [editingId, setEditingId] = useState<string | null>(null);
  const [searchText, setSearchText] = useState('');
  const [typeFilter, setTypeFilter] = useState<string>('all');

  useEffect(() => {
    fetchParameters();
  }, []);

  const fetchParameters = async () => {
    setLoading(true);
    try {
      const data = await getExtractedParameters();
      setParameters(data);
    } catch (error) {
      console.error('Failed to fetch parameters:', error);
    } finally {
      setLoading(false);
    }
  };

  const handleEdit = (id: string) => {
    setEditingId(id);
  };

  const handleSave = async (id: string, value: string) => {
    try {
      await updateParameter(id, { value });
      setEditingId(null);
      fetchParameters();
    } catch (error) {
      console.error('Failed to update parameter:', error);
    }
  };

  const handleCancel = () => {
    setEditingId(null);
  };

  const getConfidenceColor = (confidence: number) => {
    if (confidence >= 0.9) return 'green';
    if (confidence >= 0.7) return 'orange';
    return 'red';
  };

  const columns = [
    {
      title: '参数名称',
      dataIndex: 'name',
      key: 'name',
      width: '20%',
    },
    {
      title: '参数值',
      dataIndex: 'value',
      key: 'value',
      width: '20%',
      render: (text: string, record: Parameter) => {
        if (editingId === record.id) {
          return (
            <Space>
              <Input
                defaultValue={text}
                onPressEnter={(e) => handleSave(record.id, e.currentTarget.value)}
                size="small"
              />
              <CheckOutlined
                style={{ color: '#52c41a', cursor: 'pointer' }}
                onClick={(e) => handleSave(record.id, (e.currentTarget as any).value)}
              />
              <CloseOutlined
                style={{ color: '#ff4d4f', cursor: 'pointer' }}
                onClick={handleCancel}
              />
            </Space>
          );
        }
        return (
          <Space>
            <span>{text}</span>
            <EditOutlined
              style={{ cursor: 'pointer', color: '#1890ff' }}
              onClick={() => handleEdit(record.id)}
            />
          </Space>
        );
      },
    },
    {
      title: '单位',
      dataIndex: 'unit',
      key: 'unit',
      width: '10%',
    },
    {
      title: '类型',
      dataIndex: 'type',
      key: 'type',
      width: '15%',
      render: (type: string) => {
        const typeColors = {
          temperature: 'red',
          pressure: 'blue',
          flow: 'green',
          concentration: 'orange',
          other: 'default',
        };
        return <Tag color={typeColors[type] || 'default'}>{type}</Tag>;
      },
    },
    {
      title: '置信度',
      dataIndex: 'confidence',
      key: 'confidence',
      width: '15%',
      render: (confidence: number) => (
        <Tag color={getConfidenceColor(confidence)}>
          {(confidence * 100).toFixed(1)}%
        </Tag>
      ),
    },
    {
      title: '来源',
      dataIndex: 'source',
      key: 'source',
      width: '20%',
      render: (source: string) => {
        const parts = source.split('/');
        return parts[parts.length - 1];
      },
    },
  ];

  const filteredParameters = parameters.filter(param => {
    const matchesSearch = param.name.toLowerCase().includes(searchText.toLowerCase()) ||
                         param.value.toLowerCase().includes(searchText.toLowerCase());
    const matchesType = typeFilter === 'all' || param.type === typeFilter;
    return matchesSearch && matchesType;
  });

  return (
    <Card title="参数提取结果" style={{ marginBottom: 16 }}>
      <Space style={{ marginBottom: 16 }}>
        <Search
          placeholder="搜索参数名称或值"
          allowClear
          onSearch={setSearchText}
          style={{ width: 200 }}
        />
        <Select
          value={typeFilter}
          onChange={setTypeFilter}
          style={{ width: 120 }}
        >
          <Option value="all">全部类型</Option>
          <Option value="temperature">温度</Option>
          <Option value="pressure">压力</Option>
          <Option value="flow">流量</Option>
          <Option value="concentration">浓度</Option>
        </Select>
        <Button type="primary" onClick={fetchParameters}>
          刷新
        </Button>
      </Space>
      
      <Table
        columns={columns}
        dataSource={filteredParameters}
        loading={loading}
        rowKey="id"
        pagination={{
          pageSize: 20,
          showSizeChanger: true,
          showTotal: (total) => `共 ${total} 条记录`,
        }}
      />
    </Card>
  );
};

export default ParameterExtraction;
```

### 3.2 文档解析服务 (document_service.py)
```python
import os
import uuid
from datetime import datetime
from typing import List, Optional, Dict, Any
from sqlalchemy.orm import Session
from fastapi import UploadFile, HTTPException
import logging

from app.models.document import Document
from app.schemas.document import DocumentCreate, DocumentResponse
from app.utils.file_handler import save_upload_file, validate_file_format
from app.utils.document_parser import DocumentParser
from app.utils.parameter_extractor import ParameterExtractor
from app.database import get_db

logger = logging.getLogger(__name__)

class DocumentService:
    """文档处理服务类"""
    
    def __init__(self):
        self.parser = DocumentParser()
        self.extractor = ParameterExtractor()
    
    async def upload_document(
        self, 
        file: UploadFile, 
        user_id: str,
        db: Session
    ) -> DocumentResponse:
        """上传文档"""
        
        # 验证文件格式
        if not validate_file_format(file):
            raise HTTPException(status_code=400, detail="不支持的文件格式")
        
        # 生成唯一文件名
        file_extension = os.path.splitext(file.filename)[1] if file.filename else ""
        unique_filename = f"{uuid.uuid4()}{file_extension}"
        
        # 保存文件
        file_path = await save_upload_file(file, unique_filename)
        
        # 创建文档记录
        document = Document(
            id=str(uuid.uuid4()),
            filename=file.filename,
            file_path=file_path,
            file_size=os.path.getsize(file_path),
            file_type=file.content_type,
            user_id=user_id,
            status="uploaded",
            created_at=datetime.now(),
            updated_at=datetime.now()
        )
        
        db.add(document)
        db.commit()
        db.refresh(document)
        
        logger.info(f"文档上传成功: {document.filename} (ID: {document.id})")
        
        return DocumentResponse.from_orm(document)
    
    async def parse_document(
        self, 
        document_id: str, 
        user_id: str,
        db: Session
    ) -> Dict[str, Any]:
        """解析文档"""
        
        # 获取文档
        document = db.query(Document).filter(
            Document.id == document_id,
            Document.user_id == user_id
        ).first()
        
        if not document:
            raise HTTPException(status_code=404, detail="文档不存在")
        
        try:
            # 更新状态
            document.status = "parsing"
            document.updated_at = datetime.now()
            db.commit()
            
            # 解析文档
            logger.info(f"开始解析文档: {document.filename}")
            
            # 提取文本内容
            text_content = await self.parser.extract_text(document.file_path)
            
            # 提取参数
            parameters = await self.extractor.extract_parameters(
                text_content, 
                document.file_type
            )
            
            # 保存提取结果
            document.extracted_text = text_content
            document.parameters = parameters
            document.status = "parsed"
            document.parsed_at = datetime.now()
            document.updated_at = datetime.now()
            
            db.commit()
            
            logger.info(f"文档解析完成: {document.filename}, 提取参数 {len(parameters)} 个")
            
            return {
                "document_id": document_id,
                "status": "success",
                "parameters_count": len(parameters),
                "parameters": parameters
            }
            
        except Exception as e:
            # 更新错误状态
            document.status = "error"
            document.error_message = str(e)
            document.updated_at = datetime.now()
            db.commit()
            
            logger.error(f"文档解析失败: {document.filename}, 错误: {str(e)}")
            raise HTTPException(status_code=500, detail=f"文档解析失败: {str(e)}")
    
    async def get_document_parameters(
        self, 
        document_id: str, 
        user_id: str,
        db: Session
    ) -> List[Dict[str, Any]]:
        """获取文档参数"""
        
        document = db.query(Document).filter(
            Document.id == document_id,
            Document.user_id == user_id
        ).first()
        
        if not document:
            raise HTTPException(status_code=404, detail="文档不存在")
        
        if document.status != "parsed":
            raise HTTPException(status_code=400, detail="文档尚未解析")
        
        return document.parameters or []
    
    async def update_parameter(
        self, 
        document_id: str, 
        parameter_id: str,
        user_id: str,
        updates: Dict[str, Any],
        db: Session
    ) -> Dict[str, Any]:
        """更新参数"""
        
        document = db.query(Document).filter(
            Document.id == document_id,
            Document.user_id == user_id
        ).first()
        
        if not document:
            raise HTTPException(status_code=404, detail="文档不存在")
        
        parameters = document.parameters or []
        
        # 查找参数
        parameter_found = False
        for param in parameters:
            if param.get("id") == parameter_id:
                param.update(updates)
                param["updated_at"] = datetime.now().isoformat()
                parameter_found = True
                break
        
        if not parameter_found:
            raise HTTPException(status_code=404, detail="参数不存在")
        
        # 更新文档
        document.parameters = parameters
        document.updated_at = datetime.now()
        db.commit()
        
        logger.info(f"参数更新成功: 文档 {document_id}, 参数 {parameter_id}")
        
        return {"status": "success", "parameter_id": parameter_id}
    
    async def delete_document(
        self, 
        document_id: str, 
        user_id: str,
        db: Session
    ) -> bool:
        """删除文档"""
        
        document = db.query(Document).filter(
            Document.id == document_id,
            Document.user_id == user_id
        ).first()
        
        if not document:
            raise HTTPException(status_code=404, detail="文档不存在")
        
        try:
            # 删除物理文件
            if os.path.exists(document.file_path):
                os.remove(document.file_path)
            
            # 删除数据库记录
            db.delete(document)
            db.commit()
            
            logger.info(f"文档删除成功: {document.filename} (ID: {document_id})")
            return True
            
        except Exception as e:
            logger.error(f"文档删除失败: {document_id}, 错误: {str(e)}")
            raise HTTPException(status_code=500, detail="文档删除失败")
    
    async def get_user_documents(
        self, 
        user_id: str,
        skip: int = 0,
        limit: int = 100,
        db: Session
    ) -> List[DocumentResponse]:
        """获取用户文档列表"""
        
        documents = db.query(Document).filter(
            Document.user_id == user_id
        ).order_by(Document.created_at.desc()).offset(skip).limit(limit).all()
        
        return [DocumentResponse.from_orm(doc) for doc in documents]
    
    async def get_document_statistics(
        self, 
        user_id: str,
        db: Session
    ) -> Dict[str, Any]:
        """获取文档统计信息"""
        
        from sqlalchemy import func
        
        # 基础统计
        total_docs = db.query(Document).filter(Document.user_id == user_id).count()
        
        # 状态统计
        status_stats = db.query(
            Document.status,
            func.count(Document.id)
        ).filter(Document.user_id == user_id).group_by(Document.status).all()
        
        # 文件类型统计
        type_stats = db.query(
            Document.file_type,
            func.count(Document.id)
        ).filter(Document.user_id == user_id).group_by(Document.file_type).all()
        
        # 总文件大小
        total_size = db.query(func.sum(Document.file_size)).filter(
            Document.user_id == user_id
        ).scalar() or 0
        
        return {
            "total_documents": total_docs,
            "status_breakdown": dict(status_stats),
            "type_breakdown": dict(type_stats),
            "total_size_bytes": total_size,
            "total_size_mb": round(total_size / (1024 * 1024), 2)
        }

# 创建服务实例
document_service = DocumentService()
```

### 3.3 AI参数提取器 (parameter_extractor.py)
```python
import re
import json
import logging
from typing import List, Dict, Any, Optional
from datetime import datetime
import spacy
from transformers import pipeline, AutoTokenizer, AutoModelForTokenClassification
import numpy as np

logger = logging.getLogger(__name__)

class ParameterExtractor:
    """AI参数提取器"""
    
    def __init__(self):
        # 加载spaCy模型
        try:
            self.nlp = spacy.load("zh_core_web_sm")
        except OSError:
            self.nlp = spacy.load("en_core_web_sm")
        
        # 初始化参数提取模型
        self._init_parameter_models()
        
        # 定义参数模式
        self.parameter_patterns = {
            'temperature': [
                r'(\d+(?:\.\d+)?)\s*[°度](?:C|F|K)?',
                r'温度[：:]\s*(\d+(?:\.\d+)?)',
                r'T\s*=\s*(\d+(?:\.\d+)?)',
            ],
            'pressure': [
                r'(\d+(?:\.\d+)?)\s*(?:bar|atm|Pa|kPa|MPa|psi)',
                r'压力[：:]\s*(\d+(?:\.\d+)?)',
                r'P\s*=\s*(\d+(?:\.\d+)?)',
            ],
            'flow': [
                r'(\d+(?:\.\d+)?)\s*(?:m³/h|L/min|kg/h|kmol/h)',
                r'流量[：:]\s*(\d+(?:\.\d+)?)',
                r'F\s*=\s*(\d+(?:\.\d+)?)',
            ],
            'concentration': [
                r'(\d+(?:\.\d+)?)\s*(?:mol/L|wt%|ppm|ppb)',
                r'浓度[：:]\s*(\d+(?:\.\d+)?)',
                r'C\s*=\s*(\d+(?:\.\d+)?)',
            ],
        }
        
        # 单位映射
        self.unit_mappings = {
            'temperature': ['°C', '°F', 'K', '摄氏度', '华氏度', '开尔文'],
            'pressure': ['bar', 'atm', 'Pa', 'kPa', 'MPa', 'psi', '巴', '大气压', '帕斯卡'],
            'flow': ['m³/h', 'L/min', 'kg/h', 'kmol/h', '立方米每小时', '升每分钟'],
            'concentration': ['mol/L', 'wt%', 'ppm', 'ppb', '摩尔每升', '重量百分比'],
        }
    
    def _init_parameter_models(self):
        """初始化参数提取模型"""
        try:
            # 加载预训练模型
            model_name = "bert-base-chinese"
            self.tokenizer = AutoTokenizer.from_pretrained(model_name)
            self.model = AutoModelForTokenClassification.from_pretrained(model_name)
            
            # 创建NER管道
            self.ner_pipeline = pipeline(
                "ner",
                model=self.model,
                tokenizer=self.tokenizer,
                aggregation_strategy="simple"
            )
            
            logger.info("参数提取模型初始化成功")
            
        except Exception as e:
            logger.error(f"模型初始化失败: {e}")
            self.ner_pipeline = None
    
    async def extract_parameters(
        self, 
        text: str, 
        document_type: str = "unknown"
    ) -> List[Dict[str, Any]]:
        """提取参数"""
        
        logger.info(f"开始提取参数，文档类型: {document_type}")
        
        parameters = []
        
        # 1. 基于规则的参数提取
        rule_based_params = self._extract_by_rules(text)
        parameters.extend(rule_based_params)
        
        # 2. 基于NER的参数提取
        if self.ner_pipeline:
            ner_params = await self._extract_by_ner(text)
            parameters.extend(ner_params)
        
        # 3. 化学专业参数提取
        chemical_params = self._extract_chemical_parameters(text)
        parameters.extend(chemical_params)
        
        # 4. 去重和合并
        parameters = self._merge_duplicate_parameters(parameters)
        
        # 5. 验证和校准
        parameters = self._validate_parameters(parameters)
        
        # 6. 添加元数据
        for param in parameters:
            param["id"] = self._generate_parameter_id()
            param["extracted_at"] = datetime.now().isoformat()
            param["document_type"] = document_type
        
        logger.info(f"参数提取完成，共提取 {len(parameters)} 个参数")
        
        return parameters
    
    def _extract_by_rules(self, text: str) -> List[Dict[str, Any]]:
        """基于规则提取参数"""
        
        parameters = []
        
        for param_type, patterns in self.parameter_patterns.items():
            for pattern in patterns:
                matches = re.finditer(pattern, text, re.IGNORECASE)
                
                for match in matches:
                    value = match.group(1)
                    unit = self._extract_unit(match.group(0), param_type)
                    
                    # 验证数值合理性
                    if self._is_valid_value(value, param_type):
                        parameters.append({
                            "name": self._get_parameter_name(param_type),
                            "value": value,
                            "unit": unit,
                            "type": param_type,
                            "confidence": 0.8,
                            "method": "rule_based",
                            "source_text": match.group(0),
                            "position": match.span()
                        })
        
        return parameters
    
    async def _extract_by_ner(self, text: str) -> List[Dict[str, Any]]:
        """基于NER提取参数"""
        
        parameters = []
        
        try:
            # 使用NER管道
            ner_results = self.ner_pipeline(text)
            
            for entity in ner_results:
                if entity["entity_group"] in ["MISC", "NUM"]:
                    # 提取数值和单位
                    value, unit = self._parse_entity_value(entity["word"])
                    
                    if value:
                        param_type = self._infer_parameter_type(
                            entity["word"], 
                            entity["start"], 
                            text
                        )
                        
                        parameters.append({
                            "name": entity["word"],
                            "value": value,
                            "unit": unit,
                            "type": param_type,
                            "confidence": entity["score"],
                            "method": "ner",
                            "source_text": entity["word"],
                            "position": (entity["start"], entity["end"])
                        })
        
        except Exception as e:
            logger.error(f"NER提取失败: {e}")
        
        return parameters
    
    def _extract_chemical_parameters(self, text: str) -> List[Dict[str, Any]]:
        """提取化学专业参数"""
        
        parameters = []
        
        # 化学式识别
        chemical_formulas = re.findall(
            r'\b[A-Z][a-z]?\d*(?:[A-Z][a-z]?\d*)*\b', 
            text
        )
        
        # 反应条件提取
        reaction_conditions = re.findall(
            r'反应条件[：:]\s*([^。]+)',
            text
        )
        
        # 物性参数提取
        property_patterns = {
            'density': r'密度[：:]\s*(\d+(?:\.\d+)?)\s*(g/cm³|kg/m³)',
            'viscosity': r'粘度[：:]\s*(\d+(?:\.\d+)?)\s*(Pa·s|cP)',
            'boiling_point': r'沸点[：:]\s*(\d+(?:\.\d+)?)\s*[°度]C',
            'melting_point': r'熔点[：:]\s*(\d+(?:\.\d+)?)\s*[°度]C',
        }
        
        for prop_type, pattern in property_patterns.items():
            matches = re.finditer(pattern, text)
            for match in matches:
                parameters.append({
                    "name": self._get_chemical_property_name(prop_type),
                    "value": match.group(1),
                    "unit": match.group(2) if len(match.groups()) > 1 else "",
                    "type": prop_type,
                    "confidence": 0.9,
                    "method": "chemical_rule",
                    "source_text": match.group(0),
                    "position": match.span()
                })
        
        return parameters
    
    def _merge_duplicate_parameters(
        self, 
        parameters: List[Dict[str, Any]]
    ) -> List[Dict[str, Any]]:
        """合并重复参数"""
        
        merged = []
        seen = set()
        
        for param in parameters:
            # 创建唯一键
            key = f"{param['name']}:{param['value']}:{param['unit']}"
            
            if key not in seen:
                seen.add(key)
                merged.append(param)
            else:
                # 合并置信度
                existing = next(p for p in merged if 
                    f"{p['name']}:{p['value']}:{p['unit']}" == key)
                existing["confidence"] = max(
                    existing["confidence"], 
                    param["confidence"]
                )
        
        return merged
    
    def _validate_parameters(
        self, 
        parameters: List[Dict[str, Any]]
    ) -> List[Dict[str, Any]]:
        """验证参数"""
        
        validated = []
        
        for param in parameters:
            # 数值范围验证
            if self._is_value_in_range(param["value"], param["type"]):
                # 单位验证
                if self._is_valid_unit(param["unit"], param["type"]):
                    param["validated"] = True
                    validated.append(param)
                else:
                    logger.warning(f"无效单位: {param['unit']} for {param['type']}")
            else:
                logger.warning(f"数值超出范围: {param['value']} for {param['type']}")
        
        return validated
    
    def _generate_parameter_id(self) -> str:
        """生成参数ID"""
        return f"param_{datetime.now().timestamp():.0f}_{uuid.uuid4().hex[:8]}"
    
    def _extract_unit(self, text: str, param_type: str) -> str:
        """提取单位"""
        
        for unit in self.unit_mappings.get(param_type, []):
            if unit in text:
                return unit
        
        return ""
    
    def _is_valid_value(self, value: str, param_type: str) -> bool:
        """验证数值有效性"""
        
        try:
            num_value = float(value)
            
            # 根据参数类型检查范围
            ranges = {
                'temperature': (-273.15, 10000),
                'pressure': (0, 100000),
                'flow': (0, 1000000),
                'concentration': (0, 100),
            }
            
            min_val, max_val = ranges.get(param_type, (float('-inf'), float('inf')))
            return min_val <= num_value <= max_val
            
        except ValueError:
            return False
    
    def _get_parameter_name(self, param_type: str) -> str:
        """获取参数名称"""
        
        names = {
            'temperature': '温度',
            'pressure': '压力',
            'flow': '流量',
            'concentration': '浓度',
            'density': '密度',
            'viscosity': '粘度',
        }
        
        return names.get(param_type, param_type)
    
    def _get_chemical_property_name(self, prop_type: str) -> str:
        """获取化学属性名称"""
        
        names = {
            'density': '密度',
            'viscosity': '粘度',
            'boiling_point': '沸点',
            'melting_point': '熔点',
        }
        
        return names.get(prop_type, prop_type)
    
    def _parse_entity_value(self, entity_text: str) -> tuple:
        """解析实体值"""
        
        # 简单的数值和单位提取
        number_match = re.search(r'(\d+(?:\.\d+)?)', entity_text)
        unit_match = re.search(r'([a-zA-Zμ³²°]+)', entity_text)
        
        value = number_match.group(1) if number_match else None
        unit = unit_match.group(1) if unit_match else ""
        
        return value, unit
    
    def _infer_parameter_type(
        self, 
        entity_text: str, 
        start: int, 
        full_text: str
    ) -> str:
        """推断参数类型"""
        
        # 获取上下文
        context_start = max(0, start - 50)
        context_end = min(len(full_text), start + len(entity_text) + 50)
        context = full_text[context_start:context_end]
        
        # 关键词匹配
        type_keywords = {
            'temperature': ['温度', 'temperature', 'T=', '°C', '°F', 'K'],
            'pressure': ['压力', 'pressure', 'P=', 'bar', 'atm', 'Pa'],
            'flow': ['流量', 'flow', 'F=', 'm³/h', 'L/min'],
            'concentration': ['浓度', 'concentration', 'C=', 'mol/L', 'wt%'],
        }
        
        for param_type, keywords in type_keywords.items():
            if any(keyword in context for keyword in keywords):
                return param_type
        
        return 'other'
    
    def _is_value_in_range(self, value: str, param_type: str) -> bool:
        """检查数值是否在合理范围内"""
        
        return self._is_valid_value(value, param_type)
    
    def _is_valid_unit(self, unit: str, param_type: str) -> bool:
        """验证单位有效性"""
        
        valid_units = self.unit_mappings.get(param_type, [])
        return unit in valid_units or unit == ""
```
