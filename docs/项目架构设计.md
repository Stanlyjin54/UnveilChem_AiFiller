# UnveilChem 智能参数录入助手 - 前后端分离架构设计

## 1. 项目概述
在当前化工行业数字化转型浪潮中，大量工艺文档、设备规格书、P&ID图纸等非结构化数据亟待有效处理。传统人工处理方式效率低下、错误率高，严重制约了工艺优化和设备管理效率。

### 核心组件设计 1. Agent Service (代理服务)
```
# backend/app/services/agent/agent_service.py

from typing import Dict, Any, List, Optional
from dataclasses import dataclass
from enum import Enum
from pydantic import BaseModel

class TaskIntent(Enum):
    """任务意图枚举"""
    PARAMETER_EXTRACTION = "parameter_extraction"
    SIMULATION_RUN = "simulation_run"
    REPORT_GENERATION = "report_generation"
    DOCUMENT_ANALYSIS = "document_analysis"
    SOFTWARE_OPERATION = "software_operation"

class AgentRequest(BaseModel):
    """代理请求"""
    user_input: str
    context: Dict[str, Any] = {}
    attachments: List[str] = []  # 文件路径
    target_software: Optional[str] = None

class AgentResponse(BaseModel):
    """代理响应"""
    intent: TaskIntent
    confidence: float
    extracted_parameters: Dict[str, Any]
    suggested_actions: List[Dict[str, Any]]
    execution_plan: Optional[Dict[str, Any]]

class AgentService:
    """LLM代理服务"""
    
    def __init__(self):
        self.llm_client = None  # 初始化LLM客户端
        self.intent_classifier = None
        self.parameter_extractor = None
        
    async def process_request(self, request: 
    AgentRequest) -> AgentResponse:
        """处理用户请求"""
        
        # Step 1: 意图识别
        intent = await self._classify_intent
        (request.user_input)
        
        # Step 2: 参数提取
        extracted_params = await self.
        _extract_parameters(
            request.user_input,
            request.attachments,
            intent
        )
        
        # Step 3: 生成执行计划
        execution_plan = await self.
        _generate_execution_plan(
            intent,
            extracted_params,
            request.target_software
        )
        
        return AgentResponse(
            intent=intent,
            confidence=0.9,
            extracted_parameters=extracted_params,
            suggested_actions=execution_plan.get
            ("actions", []),
            execution_plan=execution_plan
        )

import json
import re

class LLMParameterExtractor:
    """基于LLM的参数提取器"""
    
    def __init__(self, llm_client):
        self.llm_client = llm_client
        self.parameter_schema = self.
        _load_parameter_schema()
        
    async def extract_from_text(self, text: str, 
    target_software: str) -> Dict[str, Any]:
        """从文本中提取参数"""
        
        prompt = f"""
        你是一个化工参数提取专家。从以下文本中提取
        {target_software}所需的参数。
        
        可识别的参数类型:
        {json.dumps(self.parameter_schema.get
        (target_software, {}), 
        ensure_ascii=False)}
        
        输入文本:
        {text}
        
        请以JSON格式返回提取的参数:
        """
        
        response = await self.llm_client.chat
        (prompt)
        return json.loads(response)
    
    async def extract_from_document(self, 
    file_path: str, target_software: str) -> Dict
    [str, Any]:
        """从文档中提取参数"""
        
        # 1. 先解析文档
        document_text = await self._parse_document
        (file_path)
        
        # 2. 使用LLM提取参数
        return await self.extract_from_text
        (document_text, target_software)

# backend/app/services/agent/tool_orchestrator.py

from typing import Dict, Any, List, Optional
from dataclasses import dataclass

@dataclass
class ToolExecution:
    """工具执行描述"""
    tool_name: str
    parameters: Dict[str, Any]
    dependencies: List[str] = None  # 依赖的其他工具
    retry_on_failure: bool = True

class ToolOrchestrator:
    """工具编排器 - 参考OpenClaw的Tool System"""
    
    def __init__(self, automation_engine):
        self.automation_engine = automation_engine
        self.available_tools = self.
        _register_tools()
        
    def _register_tools(self) -> Dict[str, Any]:
        """注册可用工具"""
        return {
            "excel": {
                "handler": self._execute_excel,
                "description": "Excel操作",
                "parameters": ["file_path", 
                "operation", "data"]
            },
            "aspen_plus": {
                "handler": self._execute_aspen,
                "description": "Aspen Plus模拟",
                "parameters": ["simulation_file", 
                "parameters"]
            },
            "dwsim": {
                "handler": self._execute_dwsim,
                "description": "DWSIM模拟",
                "parameters": ["flowsheet", 
                "parameters"]
            },
            "parameter_mapper": {
                "handler": self._map_parameters,
                "description": "参数映射转换"
            },
            "report_generator": {
                "handler": self._generate_report,
                "description": "生成报告"
            }
        }
    
    async def execute_plan(self, execution_plan: 
    Dict[str, Any]) -> Dict[str, Any]:
        """执行完整的执行计划"""
        
        results = {}
        
        for step in execution_plan.get("steps", 
        []):
            tool_name = step["tool"]
            params = step["parameters"]
            
            # 检查依赖是否满足
            if not self._check_dependencies(step.
            get("dependencies", []), results):
                return {"error": f"依赖未满足: 
                {step}"}
            
            # 执行工具
            result = await self._execute_tool
            (tool_name, params)
            results[step["id"]] = result
            
        return results

### 实施路线图 Phase 1: 基础能力建设 (2-3周)
任务 说明 LLM客户端集成 支持OpenAI/Claude/本地模型 Agent Service 意图识别 + 参数提取基础功能 工具注册系统 统一工具接口
 Phase 2: 核心功能实现 (3-4周)
任务 说明 自然语言参数解析 "将温度设为100度" → {TEMP: 100} 多软件协调 顺序/并行执行多个软件 上下文记忆 支持多轮对话
 Phase 3: 高级功能 (2-3周)
任务 说明 视觉理解 截图/UI识别 + 自动操作 复杂工作流 条件分支、循环执行 错误自恢复 自动重试 + 修复建议

这个方案的核心是在现有的自动化引擎之上增加一个 智能代理层(Agent Orchestration Layer) ，让LLM能够：

1. 理解用户意图 - 通过自然语言理解用户想要执行的任务
2. 智能提取参数 - 从文档、对话中自动提取结构化参数
3. 编排执行流程 - 自动协调多个软件适配器完成复杂任务
方案的关键设计包括：

- AgentService - LLM代理服务，处理意图识别和参数提取
- LLMParameterExtractor - 基于LLM的参数提取器
- ToolOrchestrator - 工具编排器，参考OpenClaw的Tool System设计
